Working Code

#!/usr/bin/env python import sys import os def main(): # correct number of arguments if len(sys.argv) < 3: print "error" return # command line arguments doc_folder = sys.argv[1] query = sys.argv[2] # open query with open(query) as input_file: for line in input_file: query_term = line # iterate through documents document = os.path.dirname(doc_folder) for doc in os.listdir(doc_folder): document = doc.read() word_list = document.split() count = 0 # count appearance of query term freq = word_list.count(query_term) # total word count for w in word_list: count += 1 # info we want in our output output += "\t".join(doc, query_term, freq, count) + "\n" # write output to stdout sys.stdout.write(output) main()
